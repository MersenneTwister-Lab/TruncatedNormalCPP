63  /*
64  * Monte-Carlo Integration.
65  * This function returns mean value, to get sum value multiply m
66  * to the result.
67  *
68  * @tperm I integrand function class.
69  * @tparm R Random number generator class.
70  * @tparm D Distribution class for Random number generator.
71  *
72  * @param[in] s dimension of integration area @b R.
73  * @param[in] m sample number per a trial.
74  * @param[in] N number of trials.
75  * @param[in,out] integrand integrand function class, which should have
76  * double operator()(double[]).
77  * @param[in,out] rand random number generator.
78  * @param[in,out] dist random number distribution class.
79  * @param[in] probability expected probability of returned value x is
80  * between x - absolute error and x + absolute error. this should be
81  * one of {95, 99, 999, 9999}.
82  * @return MCQMCResult.
83  */
84  template<typename I, typename R, typename D>
85  MCQMCResult monte_carlo_integration(uint32_t s,
86  uint32_t m,
87  uint32_t N,
88  I& integrand,
89  R& rand,
90  D& dist,
91  int probability = 99)
92  {
93  double point[s];
94  OnlineVariance eachintval;
95  uint32_t cnt = 0;
96  do {
97  OnlineVariance intsum;
98  for (uint32_t j = 0; j < m; ++j) {
99  for (uint32_t i = 0; i < s; ++i) {
100  point[i] = dist(rand);
101  }
102  intsum.addData(integrand(point));
103  }
104  eachintval.addData(intsum.getMean());
105  ++cnt;
106  } while ( cnt < N );
107  return MCQMCResult({eachintval.getMean(),
108  eachintval.absErr(probability)});
109  }
